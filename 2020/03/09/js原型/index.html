function partition_s(A, i, j){
    var pivot = A[j];
    var p = i,
        temp;
    for(let q = i; q < j; q++){
        if(A[q] < pivot){
            temp = A[q];
            A[q] = A[p];
            A[p] = temp;
            p +=1;
        }
    }
    A[j] = A[p];
    A[p] = pivot;
    return p;
}
