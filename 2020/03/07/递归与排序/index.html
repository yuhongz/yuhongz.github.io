function mergeSort_c(A, i, j){
    if(i >= j) return;
    var mid = Math.floor((i + j)/2);
    mergeSort_c(A, i, mid);
    mergeSort_c(A, mid+1, j);
    merge(A, i, mid, j);
}

function merge(A, i, mid, j){
    var temp = [];
    var p = i,
        q = mid+1;
    while(p <= mid && q <= j){
        if(A[p] > A[q]){
            temp.push(A[p]);
            p += 1;
        }
        if(A[p] <= A[q]){
            temp.push(A[q]);
            q += 1;
        }
    }
    if(p > mid){
        for(; q<=j; q++) temp.push(A[q])
    }
    if(q > j){
        for(; p<=mid; p++) temp.push(A[p])
    }
    for(var t = i; t <= j; t ++){
        A[t] = temp.shift();
    }
}
